import jwt from 'jsonwebtoken';
import { TokenPayload } from '../../types/auth';

export function validateToken(authHeader: string | undefined): TokenPayload {
  if (!authHeader?.startsWith('Bearer ')) {
    throw new AuthError(401, 'Missing or malformed Authorization header');
  }
  const token = authHeader.slice(7);
  try {
    return jwt.verify(token, process.env.JWT_SECRET!) as TokenPayload;
  } catch (err) {
    throw new AuthError(401, 'Invalid or expired token');
  }
}

import { db } from '../../db';

export async function lookupSession(payload: TokenPayload): Promise<SessionUser> {
  const user = await db.users.findById(payload.sub);
  if (!user || !user.active) {
    throw new AuthError(401, 'Session not found or user deactivated');
  }
  return user;
}

import { Request, Response, NextFunction } from 'express';
import { validateToken } from './validateToken';
import { lookupSession } from './lookupSession';
import { enrichRequest } from './enrichRequest';

export async function authenticateRequest(
  req: Request, res: Response, next: NextFunction
): Promise<void> {
  try {
    const payload = validateToken(req.headers.authorization);
    const user = await lookupSession(payload);
    enrichRequest(req, user, payload.sessionId);
    next();
  } catch (err) {
    if (err instanceof AuthError) {
      res.status(err.statusCode).json({ error: err.message });
    } else {
      next(err);
    }
  }
}
